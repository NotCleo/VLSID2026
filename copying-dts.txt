dump the fitImage

dumpimage -T flat_dt -p 2 -o ~/mpfs-icicle-kit-prod.dtb fitImage

dtc -I dtb -O dts -o mpfs-icicle-kit-prod.dts mpfs-icicle-kit-prod.dtb
